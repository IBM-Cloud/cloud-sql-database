CREATE TABLE CITIES(
GEONAMEID        INTEGER,
NAME             VARCHAR(200),
ASCIINAME        VARCHAR(200),
ALTERNATENAMES   VARCHAR(30000),
LATITUDE         FLOAT,
LONGITUDE        FLOAT,
FEATURECLASS     CHAR(1),
FEATURECODE      VARCHAR(10),
COUNTRYCODE      CHAR(2),
CC2              VARCHAR(200),
ADMIN1CODE       VARCHAR(20),
ADMIN2CODE       VARCHAR(80),
ADMIN3CODE       VARCHAR(20),
ADMIN4CODE       VARCHAR(20),
POPULATION       BIGINT,
ELEVATION        INTEGER,
DEM              INTEGER,
TIMEZONE         VARCHAR(40),
MODIFICATIONDATE DATE);
